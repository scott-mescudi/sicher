	dir := "folder2"
	os.Mkdir(dir, 0200)
	var wg sync.WaitGroup
	files := []string{
		"diamond.exe",
		"diamond2.exe",
		"diamond3.exe",
		"diamond4.exe",
		"diamond5.exe",
		"main.txt",
	}
	start := time.Now()
	errCh := make(chan error)
	for _, i := range files {
		fp := filepath.Join(dir, i)
		wg.Add(1)
		go func(i,fp string) {
			defer wg.Done()
			if err := copyFile(i, fp, 104_857_600); err != nil{
				errCh <-  fmt.Errorf("failed to copy %s: %v", i, err)
			}
		}(i,fp)
	}
	go func() {
		wg.Wait()
		close(errCh) 
	}()
	for err := range errCh{
		if err != nil{
			fmt.Println(err)
		}
	}
	
	elapsed := time.Since(start)
	printMemUsage()
	fmt.Printf("Time for test1: %v\n",elapsed)
